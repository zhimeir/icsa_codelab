The knockoff example below uses the CRAN knockoff package. To run it locally, you need to install the package.

library(knockoff)

fixed_equi_knockoffs <- function(X) {
  create.fixed(X, method = "equi")
}

knockoff_summary <- function(fit, active = integer()) {
  p <- length(fit$statistic)
  data.frame(
    feature = paste0("X", seq_len(p)),
    W = round(fit$statistic, 3),
    selected = seq_len(p) %in% fit$selected,
    threshold = round(fit$threshold, 3),
    truly_active = seq_len(p) %in% active,
    row.names = NULL
  )
}
set.seed(202)
n <- 120
m <- 12
active <- 1:4
q <- 0.25

X <- matrix(rnorm(n * m), n, m)
colnames(X) <- paste0("X", seq_len(m))

beta <- rep(0, m)
beta[active] <- c(1.6, 1.3, 1.0, 0.8)
y <- drop(X %*% beta + rnorm(n, sd = 0.7))

fit <- knockoff.filter(
  X = X,
  y = y,
  knockoffs = fixed_equi_knockoffs,
  statistic = stat.glmnet_coefdiff,
  fdr = q,
  offset = 1
)

results <- knockoff_summary(fit, active)

Lecture 4: Adaptive/interactive data analysis

The following uses adaptMT package implementation: adapt_glm() fits covariate-dependent null and alternative models while the procedure masks p-values and updates a threshold curve.

.libPaths(c(file.path(getwd(), "r-lib"), .libPaths()))
library(adaptMT)
library(splines)

set.seed(303)
m <- 300
q <- 0.10

x <- data.frame(x = sort(runif(m)))
signal_probability <- plogis(-4.2 + 7.8 * x$x)
is_signal <- rbinom(m, size = 1, prob = signal_probability) == 1

effect <- 1.6 + 1.5 * x$x
z <- rnorm(m, mean = ifelse(is_signal, effect, 0), sd = 1)
pvals <- pmax(pmin(1 - pnorm(z), 1 - 1e-12), 1e-12)

formulas <- paste0("ns(x, df = ", 4:6, ")")
adapt_fit <- adapt_glm(
  x = x,
  pvals = pvals,
  pi_formulas = formulas,
  mu_formulas = formulas,
  alphas = q,
  verbose = list(print = FALSE, fit = FALSE, ms = FALSE)
)

adapt_rejections <- if (is.matrix(adapt_fit$rejs)) {
  as.integer(adapt_fit$rejs[, 1])
} else {
  as.integer(unlist(adapt_fit$rejs[[1]]))
}
adapt_rejections <- adapt_rejections[!is.na(adapt_rejections)]

adapt_selected <- rep(FALSE, m)
adapt_selected[adapt_rejections] <- TRUE
bh_selected <- p.adjust(pvals, method = "BH") <= q

method_summary <- function(selected, is_signal) {
  data.frame(
    discoveries = sum(selected),
    true_discoveries = sum(selected & is_signal),
    false_discoveries = sum(selected & !is_signal),
    fdp = round(sum(selected & !is_signal) / max(1, sum(selected)), 3),
    power = round(sum(selected & is_signal) / sum(is_signal), 3)
  )
}

comparison <- rbind(
  BH = method_summary(bh_selected, is_signal),
  AdaPT = method_summary(adapt_selected, is_signal)
)
comparison
